import java.util.*;
class Pair {
    int first;
    String second;

    Pair(int f, String s) {
        first = f;
        second = s;
    }
}


Min heap on first Min on second
PriorityQueue<Pair> pq = new PriorityQueue<>(
            (a, b) -> {
                if (a.first != b.first)
                    return a.first - b.first;      
                return a.second.compareTo(b.second); 
            }
        );
